\begin{tikzpicture}[
    scale=0.88, transform shape,
    frozen/.style={rectangle, rounded corners=4pt, minimum width=3.0cm, minimum height=1.1cm,
                   align=center, draw=gray!70, fill=gray!12, very thick, font=\sffamily\small},
    trainable/.style={rectangle, rounded corners=4pt, minimum width=3.0cm, minimum height=1.1cm,
                      align=center, draw=orange!80, fill=orange!12, very thick, font=\sffamily\small},
    input/.style={rectangle, rounded corners=4pt, minimum width=3.0cm, minimum height=1.1cm,
                  align=center, draw=blue!60, fill=blue!6, thick, font=\sffamily\small},
    output/.style={rectangle, rounded corners=4pt, minimum width=3.0cm, minimum height=1.1cm,
                   align=center, draw=teal!60, fill=teal!6, thick, font=\sffamily\small},
    arr/.style={thick, ->, >=stealth, line width=1.0pt},
    darr/.style={thick, ->, >=stealth, dashed, line width=1.0pt}
]

%% ── Stage 1: Inputs ─────────────────────────────────────────────────────
\node[input] (mask) at (0, 0.9)
    {\textbf{①} Binary Mask\\\scriptsize(real or DDPM-generated)};

\node[input] (text) at (0,-0.9)
    {\textbf{②} Text Prompt\\\scriptsize\textit{"dermoscopy, benign/malignant..."}};

%% ── Stage 2: Conditioning ───────────────────────────────────────────────
\node[frozen]    (clip) at (5.0,-0.9)
    {\textbf{③} Dual CLIP Encoders\\\scriptsize$\mathbf{c}_t\!\in\!\mathbb{R}^{B\times77\times2048}$};

\node[trainable] (cn)   at (5.0, 0.9)
    {\textbf{④} ControlNet\\\scriptsize(trainable copy of U-Net enc.)};

%% ── Stage 3: Synthesis ──────────────────────────────────────────────────
\node[frozen] (unet) at (10.2, 0)
    {\textbf{⑤} SDXL U-Net Backbone\\\scriptsize(frozen, 50-step DPM-Solver++)};

\node[frozen] (vae)  at (14.0, 0)
    {\textbf{⑥} VAE Decoder\\\scriptsize(frozen, $8\times$ upsample)};

\node[output] (out)  at (17.6, 0)
    {\textbf{⑦} Synthetic\\Dermoscopy Image};

%% ── Gaussian noise input into U-Net ─────────────────────────────────────
\node[font=\sffamily\scriptsize, gray] (noise) at (10.2, 1.8)
    {$\mathbf{z}_T\!\sim\!\mathcal{N}(\mathbf{0},\mathbf{I})$ \ (initial noise)};

%% ── Connections ─────────────────────────────────────────────────────────
% Mask → ControlNet
\draw[arr] (mask.east) -- (cn.west);

% Text → CLIP
\draw[arr] (text.east) -- (clip.west);

% CLIP → U-Net (text embeddings)
\draw[arr] (clip.east) -- ++(0.3,0) |- (unet.south);

% ControlNet → U-Net (residuals, dashed orange)
\draw[darr, orange!70] (cn.east) --
    node[above, font=\sffamily\scriptsize, black]{residuals (zero-conv)}
    (unet.west|-cn);

% Noise → U-Net
\draw[arr, gray] (noise.south) -- (unet.north);

% U-Net → VAE → Output
\draw[arr] (unet) -- (vae);
\draw[arr] (vae)  -- (out);

%% ── Background stage zones ───────────────────────────────────────────────
\begin{scope}[on background layer]
    \node[fill=blue!3, draw=blue!12, rounded corners=6pt,
          fit=(mask)(text), inner sep=10pt,
          label={[font=\sffamily\scriptsize\bfseries, blue!50]above:Inputs}] {};
    \node[fill=orange!3, draw=orange!12, rounded corners=6pt,
          fit=(clip)(cn), inner sep=10pt,
          label={[font=\sffamily\scriptsize\bfseries, orange!70]above:Conditioning}] {};
    \node[fill=gray!3, draw=gray!12, rounded corners=6pt,
          fit=(unet)(vae)(out)(noise), inner sep=10pt,
          label={[font=\sffamily\scriptsize\bfseries, gray!70]above:Image Synthesis}] {};
\end{scope}
