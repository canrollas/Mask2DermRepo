\begin{tikzpicture}[

    frozen/.style={rectangle, rounded corners=4pt, minimum width=2.9cm, minimum height=1.0cm,

    trainable/.style={rectangle, rounded corners=4pt, minimum width=2.9cm, minimum height=1.0cm,
                      align=center, draw=orange!80, fill=orange!12, very thick, font=\sffamily\small},
    data/.style={rectangle, rounded corners=4pt, minimum width=2.0cm, minimum height=0.85cm,
                 align=center, draw=blue!60, fill=blue!6, thick, font=\sffamily\small},
    lossbox/.style={rectangle, rounded corners=4pt, minimum width=3.2cm, minimum height=1.0cm,
                    align=center, draw=red!60, fill=red!6, thick, font=\sffamily\small},
    arr/.style={thick, ->, >=stealth, line width=1.0pt},

]

%% ── Row 1 (y=0): Image path ─────────────────────────────────────────────
\node[data]    (img)   at (0.0,  0)   {\textbf{①}\\Input Image};
\node[frozen]  (vae)   at (3.3,  0)   {\textbf{②} VAE Encoder\\\tiny frozen};
\node[data]    (z0)    at (6.2,  0)   {$\mathbf{z}_0$};
\node[frozen]  (sched) at (9.0,  0)   {\textbf{③} Noise Scheduler\\\tiny DDPM, add $\boldsymbol{\epsilon}$ at $t$};
\node[data]    (zt)    at (12.1, 0)   {$\mathbf{z}_t$};

%% ── Row 2 (y=-2.7): Text path ───────────────────────────────────────────
\node[data]   (prom)  at (0.0, -2.7)  {\textbf{④}\\Text Prompt};
\node[frozen] (clip1) at (3.3, -2.1)  {\textbf{⑤} CLIP Enc. 1\\\tiny[B,77,768]};
\node[frozen] (clip2) at (3.3, -3.3)  {\textbf{⑥} CLIP Enc. 2\\\tiny[B,77,1280]};
\node[data]   (ct)    at (7.0, -2.7)  {$\mathbf{c}_t$\\\tiny[B,\!77,\!2048]};

%% ── Row 3 (y=-5.4): Mask path ───────────────────────────────────────────
\node[data]      (mask) at (0.0, -5.4)  {\textbf{⑦}\\Binary Mask};
\node[trainable] (cn)   at (3.8, -5.4)  {\textbf{⑧} ControlNet\\\tiny trainable};

%% ── U-Net (right column, spans all rows) ────────────────────────────────
\node[frozen, minimum height=6.2cm, minimum width=3.5cm, text width=3.2cm]
     (unet) at (15.3, -2.7)
     {\textbf{⑨}\\SDXL U-Net\\Backbone\\\tiny (frozen)\\[6pt]
      \scriptsize $\hat{\boldsymbol{\epsilon}}=\boldsymbol{\epsilon}_\theta(\mathbf{z}_t,t,\mathbf{c}_t,\mathbf{c}_s)$};

%% ── Loss ─────────────────────────────────────────────────────────────────
\node[lossbox] (loss) at (15.3, -7.2)
     {\textbf{⑩} Min-SNR Loss\\\scriptsize $w(t)\bigl\|\boldsymbol{\epsilon}-\hat{\boldsymbol{\epsilon}}\bigr\|^2$};

%% ── Connections: Row 1 ───────────────────────────────────────────────────
\draw[arr] (img)   -- (vae);
\draw[arr] (vae)   -- (z0);
\draw[arr] (z0)    -- node[above]{\tiny $+\boldsymbol{\epsilon}$} (sched);
\draw[arr] (sched) -- (zt);
\draw[arr] (zt.east) -- ++(0.5,0) |- (unet.north east);

%% ── Connections: Row 2 ───────────────────────────────────────────────────
\draw[arr] (prom.east) -- ++(0.25,0) |- (clip1.west);
\draw[arr] (prom.east) -- ++(0.25,0) |- (clip2.west);
\draw[arr] (clip1.east) -- ++(0.35,0) |- (ct.north west);
\draw[arr] (clip2.east) -- ++(0.35,0) |- (ct.south west);
\draw[arr] (ct.east) -- ++(0.5,0) |- (unet.west);

%% ── Connections: Row 3 ───────────────────────────────────────────────────

\draw[darr, orange!70] (cn.east) -- ++(1.5,0)
    node[above, xshift=0.4cm, font=\sffamily\tiny, black]{zero-conv residuals}
    |- (unet.south east);

%% ── Loss output ──────────────────────────────────────────────────────────
\draw[arr] (unet.south) -- node[right, font=\sffamily\tiny]{$\hat{\boldsymbol{\epsilon}}$} (loss.north);

%% ── Backprop (U-shaped path below everything) ────────────────────────────

\coordinate (bp2) at (cn.south |- bp1);

    (loss.south) -- (bp1)
    -- node[below, font=\sffamily\tiny, black]{backprop (ControlNet only)} (bp2)
    -- (cn.south);

          label={[font=\sffamily\scriptsize\bfseries, gray]above:Frozen Foundation Model (no gradients)}] {};
    \node[fill=orange!3, draw=orange!15, rounded corners=6pt,
          fit=(cn), inner sep=10pt,
          label={[font=\sffamily\scriptsize\bfseries, orange!70]below:Trainable}] {};
\end{scope}
